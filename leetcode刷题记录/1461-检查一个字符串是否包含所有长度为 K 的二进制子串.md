```python
class Solution:
    def hasAllCodes(self, s: str, k: int) -> bool:
        #当前可以选0，也可以选1
        n=len(s)
        hset=set()
        tmp=''
        for i in range(n):
            tmp=tmp+s[i]

            j=i-k+1
            if j<0:
                continue
            
            hset.add(s[j:i+1])
            if len(hset)==2**k:
                return True
            
            tmp=tmp[1:]
        return False
```
