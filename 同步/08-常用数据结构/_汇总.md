# 📂 08-常用数据结构

## 题目列表

```dataviewjs
const pages = dv.pages('"leetcode刷题记录"')
    .where(p => p.题号 && p.专题 === "08-常用数据结构")
    .sort(p => p.提交日期, 'desc');

const renderDiff = (d, rating) => {
  if (rating) {
    let color = "#1a7f37";
    if (rating >= 1500) color = "#9a6700";
    if (rating >= 1900) color = "#bf3989";
    if (rating >= 2400) color = "#cf222e";
    return `<span style="color:${color};font-weight:600;">${rating}</span>`;
  }
  const map = { "Easy": "简单", "Medium": "中等", "Hard": "困难" };
  return map[d] || d;
};
const renderMastery = m => "⭐".repeat(Number(m) || 0);
const renderElapsed = e => {
  if (!e) return "-";
  const m = Number(e);
  return m >= 60 ? `${Math.floor(m/60)}h${m%60}m` : `${m}min`;
};
const renderTitle = (p) => {
  const externalLink = p.url ? ` <a href="${p.url}" target="_blank" style="color:#888;font-size:11px;text-decoration:none;">↗</a>` : '';
  return dv.span(`${p.file.link}${externalLink}`);
};

dv.table(
  ["题目", "关键词", "难度", "熟练度", "耗时", "日期"],
  pages.map(p => [
    renderTitle(p),
    p.关键词 || "",
    dv.span(renderDiff(p.难度, p.难度分)),
    renderMastery(p.掌握程度),
    renderElapsed(p.耗时),
    p.提交日期 ? p.提交日期.toString().slice(0, 10) : "",
  ])
);
```
