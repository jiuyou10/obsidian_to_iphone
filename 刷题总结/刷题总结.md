# 📚 LeetCode 刷题总结

> 💡 点**题目文字**跳到笔记看完整总结和代码,点 **↗** 跳 LeetCode 原题

## 📋 全部题目

```dataviewjs
const pages = dv.pages('"leetcode刷题记录"').where(p => p.题号).sort(p => p.提交日期, 'desc');

const renderDiff = (d, rating) => {
  if (rating) {
    let color = "#1a7f37";
    if (rating >= 1500) color = "#9a6700";
    if (rating >= 1900) color = "#bf3989";
    if (rating >= 2400) color = "#cf222e";
    return `<span style="color:${color};font-weight:600;font-variant-numeric:tabular-nums;">${rating}</span>`;
  }
  const map = { "Easy": "简单", "Medium": "中等", "Hard": "困难" };
  return map[d] || d;
};

const renderMastery = m => "⭐".repeat(Number(m) || 0);

const renderElapsed = e => {
  if (!e) return "-";
  const m = Number(e);
  return m >= 60 ? `${Math.floor(m/60)}h${m%60}m` : `${m}min`;
};

const renderTag = t => {
  if (!t) return "";
  const tags = Array.isArray(t) ? t : [t];
  const colors = {
    "数组": "#fff3d6", "字符串": "#ffe6d6", "链表": "#d6f4ff",
    "二叉树": "#d6ffe0", "哈希表": "#fde2e2", "栈与队列": "#e8d6ff",
    "动态规划": "#ffd6e8", "贪心": "#d6ffd6", "双指针": "#d6e6ff",
    "回溯": "#ffd6f0", "图": "#d6fff0", "topK": "#ffe6c4",
    "数学": "#fff3d6", "字符串匹配": "#ffe6d6", "二分查找": "#d6e6ff",
    "排序": "#e8d6ff", "堆": "#fde2e2", "前缀和": "#d6ffe0",
    "位运算": "#e8d6ff",
  };
  return tags.slice(0, 2).map(tag => {
    const c = colors[tag] || "#e8e8e8";
    return `<span style="background:${c};padding:2px 10px;border-radius:10px;font-size:11px;color:#333;margin-right:3px;">${tag}</span>`;
  }).join("");
};

const renderTitle = (p) => {
  const externalLink = p.url ? ` <a href="${p.url}" target="_blank" style="color:#888;font-size:11px;text-decoration:none;" title="在 LeetCode 打开">↗</a>` : '';
  return dv.span(`${p.file.link}${externalLink}`);
};

dv.table(
  ["题目", "关键词", "分类", "难度", "熟练度", "耗时", "日期"],
  pages.map(p => [
    renderTitle(p),
    p.关键词 || "",
    dv.span(renderTag(p.标签)),
    dv.span(renderDiff(p.难度, p.难度分)),
    renderMastery(p.掌握程度),
    renderElapsed(p.耗时),
    p.提交日期 ? p.提交日期.toString().slice(0, 10) : "",
  ])
);
```

## 📊 难度分布

```dataview
TABLE WITHOUT ID
  length(rows) AS "题数"
FROM "leetcode刷题记录"
WHERE 题号
GROUP BY 难度 AS 难度
```
